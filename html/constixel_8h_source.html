53[[nodiscard]] static constexpr float fast_exp2(const float p) {
54 const float offset = (p < 0) ? 1.0f : 0.0f;
55 const float clipp = (p < -126) ? -126.0f : p;
56 const float z = clipp - static_cast<float>(static_cast<int32_t>(clipp)) + offset;
57 return std::bit_cast<float>(
58 static_cast<uint32_t>((1 << 23) * (clipp + 121.2740575f + 27.7280233f / (4.84252568f - z) - 1.49012907f * z)));
59}

61[[nodiscard]] static constexpr float fast_log2(const float x) {
62 uint32_t xi = std::bit_cast<uint32_t>(x);
63 float xf = std::bit_cast<float>((xi & 0x007FFFFF) | 0x3f000000);
64 const float y = static_cast<float>(xi) * 1.1920928955078125e-7f;
65 return y - 124.22551499f - 1.498030302f * xf - 1.72587999f / (0.3520887068f + xf);
66}

68static constexpr float fast_sqrtf(const float x) {
69 int32_t i = std::bit_cast<int>(x);
70 int k = i & 0x00800000;
71 float y;
72 if (k != 0) {
73 i = 0x5ed9d098 - (i >> 1);
74 y = std::bit_cast<float>(i);
75 y = 2.33139729f * y * ((-x * y * y) + 1.07492042f);
76 } else {
77 i = 0x5f19d352 - (i >> 1);
78 y = std::bit_cast<float>(i);
79 y = 0.82420468f * y * ((-x * y * y) + 2.14996147f);
80 }
81 float c = x * y;
82 float r = ((y * -c) + 1.0f);
83 y = ((0.5f * c * r) + c);
84 return y;
85}

87[[nodiscard]] static constexpr float fast_pow(const float x, const float p) {
88 return fast_exp2(p * fast_log2(x));

93[[nodiscard]] static consteval double consteval_cos(double x, int32_t terms = 10) {
94 x = x - 6.283185307179586 * static_cast<int32_t>(x / 6.283185307179586); // wrap x to [0, 2π)
95 double res = 1.0, term = 1.0;
96 double x2 = x * x;
97 for (int32_t i = 1; i < terms; ++i) {
98 term *= -x2 / ((2 * i - 1) * (2 * i));
99 res += term;
100 }
101 return res;
102}
103
104[[nodiscard]] static consteval double consteval_sin(double x, int32_t terms = 10) {
105 x = x - 6.283185307179586 * static_cast<int32_t>(x / 6.283185307179586); // wrap x to [0, 2π)
106 double res = x, term = x;
107 double x2 = x * x;
108 for (int32_t i = 1; i < terms; ++i) {
109 term *= -x2 / ((2 * i) * (2 * i + 1));
110 res += term;
111 }
112 return res;
113}

127[[nodiscard]] static constexpr float linear_to_srgb(float c) {
128 if (c <= 0.0031308f) {
129 return 12.92f * c;
130 } else {
131 if (std::is_constant_evaluated()) {
132 return 1.055f * fast_pow(c, 1.0f / 2.4f) - 0.055f;
133 } else {
134 return 1.055f * std::pow(c, 1.0f / 2.4f) - 0.055f;
135 }
136 }
137}

139[[nodiscard]] static constexpr float srgb_to_linear(float s) {
140 if (s <= 0.040449936f) {
141 return s / 12.92f;
142 } else {
143 if (std::is_constant_evaluated()) {
144 return fast_pow((s + 0.055f) / 1.055f, 2.4f);
145 } else {
146 return std::pow((s + 0.055f) / 1.055f, 2.4f);
147 }
148 }

151[[nodiscard]] static consteval srgb oklab_to_srgb_consteval(const oklab &oklab) {
152 double l = oklab.l;
153 double a = oklab.a;
154 double b = oklab.b;
155
156 double l_ = l + 0.3963377774 * a + 0.2158037573 * b;
157 double m_ = l - 0.1055613458 * a - 0.0638541728 * b;
158 double s_ = l - 0.0894841775 * a - 1.2914855480 * b;
159
160 double r = 4.0767416621 * l_ - 3.3077115913 * m_ + 0.2309699292 * s_;
161 double g = -1.2684380046 * l_ + 2.6097574011 * m_ - 0.3413193965 * s_;
162 double bl = -0.0041960863 * l_ - 0.7034186168 * m_ + 1.7076147031 * s_;
163
164 return {static_cast<double>(linear_to_srgb(std::max(0.0f, std::min(1.0f, static_cast<float>(r))))),
165 static_cast<double>(linear_to_srgb(std::max(0.0f, std::min(1.0f, static_cast<float>(g))))),
166 static_cast<double>(linear_to_srgb(std::max(0.0f, std::min(1.0f, static_cast<float>(bl)))))};
167}

225 for (size_t j = 0; j < 4; j++) {
226 linearpal_neon.at(i * 3 + j) = linearpal.at((i + j) * 3 + 0);
227 linearpal_neon.at(i * 3 + j + 4) = linearpal.at((i + j) * 3 + 1);
228 linearpal_neon.at(i * 3 + j + 8) = linearpal.at((i + j) * 3 + 2);
229 }
230 }
231 }
232#endif // #if defined(__ARM_NEON)
233#if defined(__AVX2__)
234 if (pal.size() >= 8) {
235 for (size_t i = 0; i < pal.size(); i += 8) {
236 for (size_t j = 0; j < 8; j++) {
237 linearpal_avx2.at(i * 3 + j) = linearpal.at((i + j) * 3 + 0);
238 linearpal_avx2.at(i * 3 + j + 8) = linearpal.at((i + j) * 3 + 1);

246 [[nodiscard]] constexpr uint8_t nearest(int32_t r, int32_t g, int32_t b) const {
247 int32_t best = 0, bestd = 1UL << 30;
248 for (size_t i = 0; i < pal.size(); ++i) {
249 int32_t dr = r - static_cast<int32_t>((pal[i] >> 16) & 0xFF);
250 int32_t dg = g - static_cast<int32_t>((pal[i] >> 8) & 0xFF);
251 int32_t db = b - static_cast<int32_t>((pal[i] >> 0) & 0xFF);
252 int32_t d = dr * dr + dg * dg + db * db;
253 if (d < bestd) {
254 bestd = d;
255 best = static_cast<int32_t>(i);
256 }
257 }
258 return static_cast<uint8_t>(best);
259 }

269 [[nodiscard]] constexpr uint8_t nearest_linear(float r, float g, float b) const {
270#if defined(__ARM_NEON)
271 if (!std::is_constant_evaluated() && pal.size() >= 4) {
272 const float32x4_t vR = vdupq_n_f32(r);
273 const float32x4_t vG = vdupq_n_f32(g);
274 const float32x4_t vB = vdupq_n_f32(b);
275
276 float best = std::numeric_limits<float>::infinity();
277 std::size_t bestIdx = 0;
278
279 for (size_t i = 0; i < pal.size(); i += 4) {
280 float32x4_t dr = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3]), vR);
281 float32x4_t dg = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3 + 4]), vG);
282 float32x4_t db = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3 + 8]), vB);
283
284#if defined(__aarch64__) && defined(__ARM_FEATURE_FMA)
285 float32x4_t dist = vfmaq_f32(vfmaq_f32(vmulq_f32(dr, dr), dg, dg), db, db);
286#else
287 float32x4_t dist = vaddq_f32(vaddq_f32(vmulq_f32(dr, dr), vmulq_f32(dg, dg)), vmulq_f32(db, db));
288#endif
289
290 float32x2_t lo = vget_low_f32(dist), hi = vget_high_f32(dist);
291 float d0 = vget_lane_f32(lo, 0);
292 if (d0 < best) {
293 best = d0;
294 bestIdx = i;
295 }
296 float d1 = vget_lane_f32(lo, 1);
297 if (d1 < best) {
298 best = d1;
299 bestIdx = i + 1;
300 }
301 float d2 = vget_lane_f32(hi, 0);
302 if (d2 < best) {
303 best = d2;
304 bestIdx = i + 2;
305 }
306 float d3 = vget_lane_f32(hi, 1);
307 if (d3 < best) {
308 best = d3;
309 bestIdx = i + 3;
310 }
311 }
312 return static_cast<uint8_t>(bestIdx);
313 }
314#endif // #if defined(__ARM_NEON)
315#if defined(__AVX2__)
316 if (!std::is_constant_evaluated() && pal.size() >= 8) {
317 const __m256 vR = _mm256_set1_ps(r);
318 const __m256 vG = _mm256_set1_ps(g);
319 const __m256 vB = _mm256_set1_ps(b);
320
321 float best = std::numeric_limits<float>::infinity();
322 std::uint8_t bestIdx = 0;
323
324 for (size_t i = 0; i < pal.size(); i += 8) {
325 __m256 pr = _mm256_load_ps(&linearpal_avx2[i * 3]);
326 __m256 pg = _mm256_load_ps(&linearpal_avx2[i * 3 + 8]);
327 __m256 pb = _mm256_load_ps(&linearpal_avx2[i * 3 + 16]);
328
329 __m256 dist =
330 _mm256_fmadd_ps(_mm256_sub_ps(pr, vR), _mm256_sub_ps(pr, vR),
331 _mm256_fmadd_ps(_mm256_sub_ps(pg, vG), _mm256_sub_ps(pg, vG),
332 _mm256_mul_ps(_mm256_sub_ps(pb, vB), _mm256_sub_ps(pb, vB))));
333
334 alignas(32) float d[8];
335 _mm256_store_ps(d, dist);
336
337 for (size_t lane = 0; lane < 8; lane++) {
338 if (d[lane] < best) {
339 best = d[lane];
340 bestIdx = static_cast<uint8_t>(i + lane);
341 }
342 }
343 }
344
345 return static_cast<uint8_t>(bestIdx);
346 }
347#endif // #if defined(__AVX2__)
348 size_t best = 0;
349 float bestd = 100.0f;
350 for (size_t i = 0; i < pal.size(); ++i) {
351 float dr = r - linearpal.at(i * 3 + 0);
352 float dg = g - linearpal.at(i * 3 + 1);
353 float db = b - linearpal.at(i * 3 + 2);
354 float d = dr * dr + dg * dg + db * db;
355 if (d < bestd) {
356 bestd = d;
357 best = i;
358 }
359 }
360 return static_cast<uint8_t>(best);
361 }
362};

368template <size_t N, typename T>
369class hextree {
370 static constexpr T bitslices = ((sizeof(T) * 8) / 4) - 1;
371
372 static constexpr size_t child_nodes_n = 1UL << 4;
373 static constexpr size_t child_nodes_n_mask = child_nodes_n - 1;
374
375 struct node {
376 T child[child_nodes_n]{};
377 constexpr node() {
378 for (auto &c : child) {
379 c = invalid;
380 }
381 }
382 };
383
384 public:
385 static constexpr T invalid = std::numeric_limits<T>::max();
386
387 std::array<node, N> nodes{};
388
389 [[nodiscard]] consteval size_t byte_size() const {
390 return sizeof(node) * nodes.size();
391 }
392
393 hextree(const hextree &) = delete;
394 hextree &operator=(const hextree &) = delete;
395
396 explicit consteval hextree() {
397 }
398
399 template <std::size_t NS>
400 explicit consteval hextree(const std::array<std::pair<T, T>, NS> &in) {
401 nodes[0] = node{};
402 T node_cnt = 1;
403 for (auto [key, val] : in) {
404 T idx = 0;
405 for (T d = 0; d < bitslices; d++) {
406 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
407 T next = nodes.at(idx).child[nib];
408 if (next == invalid) {
409 next = node_cnt;
410 nodes.at(idx).child[nib] = next;
411 node_cnt++;
412 }
413 idx = next;
414 }
415 nodes.at(idx).child[key & child_nodes_n_mask] = val;
416 }
417 }
418
419 template <std::size_t NS>
420 [[nodiscard]] static consteval T size(const std::array<std::pair<T, T>, NS> &in) {
421 std::vector<node> vnodes{1};
422 vnodes.assign(1, node{});
423 for (auto [key, val] : in) {
424 T idx = 0;
425 for (T d = 0; d < bitslices; d++) {
426 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
427 T next = vnodes.at(idx).child[nib];
428 if (next == invalid) {
429 next = static_cast<T>(vnodes.size());
430 vnodes.at(idx).child[nib] = next;
431 vnodes.emplace_back();
432 }
433 idx = next;
434 }
435 vnodes.at(idx).child[key & child_nodes_n_mask] = val;
436 }
437 return static_cast<T>(vnodes.size());
438 }
439
440 [[nodiscard]] constexpr T lookup(T key) const {
441 T idx = 0;
442 for (T d = 0; d < bitslices; ++d) {
443 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
444 T next = nodes.at(idx).child[nib];
445 if (next == invalid) {
446 return invalid;
447 }
448 idx = next;
449 }
450 return nodes.at(idx).child[key & child_nodes_n_mask];
451 }

587 static constexpr void png_palette(F &&char_out, const P &palette) {
588 std::array<char, 256 * 3 + 4> header;
589 size_t i = 0;
590 header.at(i++) = 'P';
591 header.at(i++) = 'L';
592 header.at(i++) = 'T';
593 header.at(i++) = 'E';

636 const size_t extra_data = size_t{24};

640 size_t i = 0;
641 header.at(i++) = 'I';

648 header.at(i++) = (((bytes_to_copy + 1) >> 0) & 0xFF);
649 header.at(i++) = (((bytes_to_copy + 1) >> 8) & 0xFF);

656 header.at(i++) = line[c];

661 png_write_array(char_out, header, i);
662 png_write_crc32(char_out, header, i);
663
664 bytes -= bytes_to_copy;
665 }
666 return adler32_sum;

754 template <typename PBT, size_t PBS>
755 struct palette_bitset {
756 constexpr void mark(PBT col) {
757 PBT idx = (col >> 5) & set_idx_mask;
758 set[idx] |= 1UL << (col & 0x1F);
759 }
760
761 constexpr void clear() {
762 set.fill(0);
763 }

765 [[nodiscard]] constexpr size_t genstack(std::array<PBT, PBS> &stack) const {
766 size_t count = 0;
767 for (size_t c = 0; c < PBS / 32; c++) {
768 for (size_t d = 0; d < 32; d++) {
769 if ((set[c] >> d) & 1) {
770 stack.at(count++) = static_cast<PBT>(c * 32 + d);
771 }
772 }
773 }
774 return count;
775 }
776
777 std::array<uint32_t, PBS / 32> set{};
778 static constexpr size_t set_idx_mask = (1UL << sizeof(set) / 4) - 1;
779 };

876 static constexpr void transpose8x8(std::array<uint8_t, 8> &a) {
877 // clang-format off
878 uint32_t x = (static_cast<uint32_t>(a[0]) << 24) |
879 (static_cast<uint32_t>(a[1]) << 16) |
880 (static_cast<uint32_t>(a[2]) << 8) |
881 (static_cast<uint32_t>(a[3]));
882 uint32_t y = (static_cast<uint32_t>(a[4]) << 24) |
883 (static_cast<uint32_t>(a[5]) << 16) |
884 (static_cast<uint32_t>(a[6]) << 8) |
885 (static_cast<uint32_t>(a[7]));
886
887 uint32_t t = (x ^ (x >> 7)) & 0x00AA00AA; x = x ^ t ^ (t << 7);
888 t = (y ^ (y >> 7)) & 0x00AA00AA; y = y ^ t ^ (t << 7);
889 t = (x ^ (x >> 14)) & 0x0000CCCC; x = x ^ t ^ (t << 14);
890 t = (y ^ (y >> 14)) & 0x0000CCCC; y = y ^ t ^ (t << 14);
891
892 t = ((y >> 4) & 0x0F0F0F0F) | (x & 0xF0F0F0F0);
893 y = ((x << 4) & 0xF0F0F0F0) | (y & 0x0F0F0F0F);
894
895 a[0] = static_cast<uint8_t>(t >> 24);
896 a[1] = static_cast<uint8_t>(t >> 16);
897 a[2] = static_cast<uint8_t>(t >> 8);
898 a[3] = static_cast<uint8_t>(t);
899 a[4] = static_cast<uint8_t>(y >> 24);
900 a[5] = static_cast<uint8_t>(y >> 16);
901 a[6] = static_cast<uint8_t>(y >> 8);
902 a[7] = static_cast<uint8_t>(y);
903 // clang-format on
904 }

906 static constexpr void transpose(const uint8_t *src, uint8_t *dst) {
907 std::array<uint8_t, 8> tmp;
908 size_t src_stride = ((W + 7) / 8);
909 size_t dst_stride = ((H + 7) / 8);
910 for (size_t y = 0; y < dst_stride; y++) {
911 for (size_t x = 0; x < src_stride; x++) {
912 size_t xl = std::min(size_t{8}, H - (y * 8));
913 for (size_t c = 0; c < xl; c++) {
914 tmp[c] = src[(y * 8 + c) * src_stride + x];
915 }
916 for (size_t c = xl; c < 8; c++) {
917 tmp[c] = 0;
918 }
919 transpose8x8(tmp);
920 size_t yl = std::min(size_t{8}, W - (x * 8));
921 for (size_t c = 0; c < yl; c++) {
922 dst[(x * 8 + c) * dst_stride + y] = tmp[c];
923 }
924 }
925 }

943 static constexpr void span(std::array<uint8_t, image_size> &data, size_t xl0, size_t xr0, size_t y, uint8_t col) {
944 col &= (1UL << bits_per_pixel) - 1;
945 size_t xl8 = xl0 / 8;
946 xl0 %= 8;
947 size_t xr8 = xr0 / 8;
948 xr0 %= 8;
949 size_t xs8 = xr8 - xl8;
950 uint8_t c8 =
951 static_cast<uint8_t>(col << 7 | col << 6 | col << 5 | col << 4 | col << 3 | col << 2 | col << 1 | col << 0);
952 constexpr uint8_t ml[] = {0b11111111, 0b01111111, 0b00111111, 0b00011111,
953 0b00001111, 0b00000111, 0b00000011, 0b00000001};
954 constexpr uint8_t mr[] = {0b00000000, 0b10000000, 0b11000000, 0b11100000,
955 0b11110000, 0b11111000, 0b11111100, 0b11111110};
956 uint8_t *yptr = &data.data()[y * bytes_per_line];
957 if (xs8 > 0) {
958 yptr[xl8] &= ~ml[xl0];
959 yptr[xl8] |= ml[xl0] & c8;
960 for (size_t x = xl8 + 1; x < xr8; x++) {
961 yptr[x] = c8;
962 }
963 yptr[xr8] &= ~mr[xr0];
964 yptr[xr8] |= mr[xr0] & c8;
965 } else {
966 yptr[xl8] &= ~(ml[xl0] & mr[xr0]);
967 yptr[xl8] |= (ml[xl0] & mr[xr0] & c8);
968 }
969 }
