53[[nodiscard]] static constexpr float fast_exp2(const float p) {
54 const float offset = (p < 0) ? 1.0f : 0.0f;
55 const float clipp = (p < -126) ? -126.0f : p;
56 const float z = clipp - static_cast<float>(static_cast<int32_t>(clipp)) + offset;
57 return std::bit_cast<float>(
58 static_cast<uint32_t>((1 << 23) * (clipp + 121.2740575f + 27.7280233f / (4.84252568f - z) - 1.49012907f * z)));
59}

61[[nodiscard]] static constexpr float fast_log2(const float x) {
62 uint32_t xi = std::bit_cast<uint32_t>(x);
63 float xf = std::bit_cast<float>((xi & 0x007FFFFF) | 0x3f000000);
64 const float y = static_cast<float>(xi) * 1.1920928955078125e-7f;
65 return y - 124.22551499f - 1.498030302f * xf - 1.72587999f / (0.3520887068f + xf);
66}

68static constexpr float fast_sqrtf(const float x) {
69 int32_t i = std::bit_cast<int>(x);
70 int k = i & 0x00800000;
71 float y;
72 if (k != 0) {
73 i = 0x5ed9d098 - (i >> 1);
74 y = std::bit_cast<float>(i);
75 y = 2.33139729f * y * ((-x * y * y) + 1.07492042f);
76 } else {
77 i = 0x5f19d352 - (i >> 1);
78 y = std::bit_cast<float>(i);
79 y = 0.82420468f * y * ((-x * y * y) + 2.14996147f);
80 }
81 float c = x * y;
82 float r = ((y * -c) + 1.0f);
83 y = ((0.5f * c * r) + c);
84 return y;
85}

87[[nodiscard]] static constexpr float fast_pow(const float x, const float p) {
88 return fast_exp2(p * fast_log2(x));

93[[nodiscard]] static consteval double cos_consteval(double x, int32_t terms = 10) {
94 x = x - 6.283185307179586 * static_cast<int32_t>(x / 6.283185307179586); // wrap x to [0, 2π)
95 double res = 1.0, term = 1.0;
96 double x2 = x * x;
97 for (int32_t i = 1; i < terms; ++i) {
98 term *= -x2 / ((2 * i - 1) * (2 * i));
99 res += term;
100 }
101 return res;
102}
103
104[[nodiscard]] static consteval double sin_consteval(double x, int32_t terms = 10) {
105 x = x - 6.283185307179586 * static_cast<int32_t>(x / 6.283185307179586); // wrap x to [0, 2π)
106 double res = x, term = x;
107 double x2 = x * x;
108 for (int32_t i = 1; i < terms; ++i) {
109 term *= -x2 / ((2 * i) * (2 * i + 1));
110 res += term;
111 }
112 return res;
113}

115[[nodiscard]] static consteval double pow_consteval(double base, double exp, int32_t terms = 10) {
116 if (base <= 0.0)
117 return (base == 0.0) ? 0.0 : 0.0 / 0.0; // NaN for negative base
118 double ln = 0.0, y = (base - 1) / (base + 1);
119 double y2 = y * y, num = y;
120 for (int32_t i = 1; i <= terms; ++i) {
121 ln += num / (2 * i - 1);
122 num *= y2;
123 }
124 ln *= 2;
125 double res = 1.0, term = 1.0, x = exp * ln;
126 for (int32_t i = 1; i < terms; ++i) {
127 term *= x / i;
128 res += term;
129 }
130 return res;
131}

185[[nodiscard]] static consteval srgb oklab_to_srgb_consteval(const oklab &oklab) {
186 double l = oklab.l;
187 double a = oklab.a;
188 double b = oklab.b;
189
190 double l_ = l + 0.3963377774 * a + 0.2158037573 * b;
191 double m_ = l - 0.1055613458 * a - 0.0638541728 * b;
192 double s_ = l - 0.0894841775 * a - 1.2914855480 * b;
193
194 double r = 4.0767416621 * l_ - 3.3077115913 * m_ + 0.2309699292 * s_;
195 double g = -1.2684380046 * l_ + 2.6097574011 * m_ - 0.3413193965 * s_;
196 double bl = -0.0041960863 * l_ - 0.7034186168 * m_ + 1.7076147031 * s_;
197
198 return {linear_to_srgb_consteval(std::max(0.0, std::min(1.0, r))),
199 linear_to_srgb_consteval(std::max(0.0, std::min(1.0, g))),
200 linear_to_srgb_consteval(std::max(0.0, std::min(1.0, bl)))};

270 [[nodiscard]] constexpr uint8_t nearest(int32_t r, int32_t g, int32_t b) const {
271 int32_t best = 0, bestd = 1UL << 30;
272 for (size_t i = 0; i < pal.size(); ++i) {
273 int32_t dr = r - static_cast<int32_t>((pal[i] >> 16) & 0xFF);
274 int32_t dg = g - static_cast<int32_t>((pal[i] >> 8) & 0xFF);
275 int32_t db = b - static_cast<int32_t>((pal[i] >> 0) & 0xFF);
276 int32_t d = dr * dr + dg * dg + db * db;
277 if (d < bestd) {
278 bestd = d;
279 best = static_cast<int32_t>(i);
280 }
281 }
282 return static_cast<uint8_t>(best);
283 }

293 [[nodiscard]] constexpr uint8_t nearest_linear(float r, float g, float b) const {
294#if defined(__ARM_NEON)
295 if (!std::is_constant_evaluated() && pal.size() >= 4) {
296 const float32x4_t vR = vdupq_n_f32(r);
297 const float32x4_t vG = vdupq_n_f32(g);
298 const float32x4_t vB = vdupq_n_f32(b);
299
300 float best = std::numeric_limits<float>::infinity();
301 std::size_t bestIdx = 0;
302
303 for (size_t i = 0; i < pal.size(); i += 4) {
304 float32x4_t dr = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3]), vR);
305 float32x4_t dg = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3 + 4]), vG);
306 float32x4_t db = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3 + 8]), vB);
307
308#if defined(__aarch64__) && defined(__ARM_FEATURE_FMA)
309 float32x4_t dist = vfmaq_f32(vfmaq_f32(vmulq_f32(dr, dr), dg, dg), db, db);
310#else
311 float32x4_t dist = vaddq_f32(vaddq_f32(vmulq_f32(dr, dr), vmulq_f32(dg, dg)), vmulq_f32(db, db));
312#endif
313
314 float32x2_t lo = vget_low_f32(dist), hi = vget_high_f32(dist);
315 float d0 = vget_lane_f32(lo, 0);
316 if (d0 < best) {
317 best = d0;
318 bestIdx = i;
319 }
320 float d1 = vget_lane_f32(lo, 1);
321 if (d1 < best) {
322 best = d1;
323 bestIdx = i + 1;
324 }
325 float d2 = vget_lane_f32(hi, 0);
326 if (d2 < best) {
327 best = d2;
328 bestIdx = i + 2;
329 }
330 float d3 = vget_lane_f32(hi, 1);
331 if (d3 < best) {
332 best = d3;
333 bestIdx = i + 3;
334 }
335 }
336 return static_cast<uint8_t>(bestIdx);
337 }
338#endif // #if defined(__ARM_NEON)
339#if defined(__AVX2__)
340 if (!std::is_constant_evaluated() && pal.size() >= 8) {
341 const __m256 vR = _mm256_set1_ps(r);
342 const __m256 vG = _mm256_set1_ps(g);
343 const __m256 vB = _mm256_set1_ps(b);
344
345 float best = std::numeric_limits<float>::infinity();
346 std::uint8_t bestIdx = 0;
347
348 for (size_t i = 0; i < pal.size(); i += 8) {
349 __m256 pr = _mm256_load_ps(&linearpal_avx2[i * 3]);
350 __m256 pg = _mm256_load_ps(&linearpal_avx2[i * 3 + 8]);
351 __m256 pb = _mm256_load_ps(&linearpal_avx2[i * 3 + 16]);
352
353 __m256 dist =
354 _mm256_fmadd_ps(_mm256_sub_ps(pr, vR), _mm256_sub_ps(pr, vR),
355 _mm256_fmadd_ps(_mm256_sub_ps(pg, vG), _mm256_sub_ps(pg, vG),
356 _mm256_mul_ps(_mm256_sub_ps(pb, vB), _mm256_sub_ps(pb, vB))));
357
358 alignas(32) float d[8];
359 _mm256_store_ps(d, dist);
360
361 for (size_t lane = 0; lane < 8; lane++) {
362 if (d[lane] < best) {
363 best = d[lane];
364 bestIdx = static_cast<uint8_t>(i + lane);
365 }
366 }
367 }
368
369 return static_cast<uint8_t>(bestIdx);
370 }
371#endif // #if defined(__AVX2__)
372 size_t best = 0;
373 float bestd = 100.0f;
374 for (size_t i = 0; i < pal.size(); ++i) {
375 float dr = r - linearpal.at(i * 3 + 0);
376 float dg = g - linearpal.at(i * 3 + 1);
377 float db = b - linearpal.at(i * 3 + 2);
378 float d = dr * dr + dg * dg + db * db;
379 if (d < bestd) {
380 bestd = d;
381 best = i;
382 }
383 }
384 return static_cast<uint8_t>(best);
385 }

392template <size_t N, typename T>
393class hextree {
394 static constexpr T bitslices = ((sizeof(T) * 8) / 4) - 1;
395
396 static constexpr size_t child_nodes_n = 1UL << 4;
397 static constexpr size_t child_nodes_n_mask = child_nodes_n - 1;
398
399 struct node {
400 T child[child_nodes_n]{};
401 constexpr node() {
402 for (auto &c : child) {
403 c = invalid;
404 }
405 }
406 };
407
408 public:
409 static constexpr T invalid = std::numeric_limits<T>::max();
410
411 std::array<node, N> nodes{};
412
413 [[nodiscard]] consteval size_t byte_size() const {
414 return sizeof(node) * nodes.size();
415 }
416
417 hextree(const hextree &) = delete;
418 hextree &operator=(const hextree &) = delete;
419
420 explicit consteval hextree() {
421 }
422
423 template <std::size_t NS>
424 explicit consteval hextree(const std::array<std::pair<T, T>, NS> &in) {
425 nodes[0] = node{};
426 T node_cnt = 1;
427 for (auto [key, val] : in) {
428 T idx = 0;
429 for (T d = 0; d < bitslices; d++) {
430 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
431 T next = nodes.at(idx).child[nib];
432 if (next == invalid) {
433 next = node_cnt;
434 nodes.at(idx).child[nib] = next;
435 node_cnt++;
436 }
437 idx = next;
438 }
439 nodes.at(idx).child[key & child_nodes_n_mask] = val;
440 }
441 }
442
443 template <std::size_t NS>
444 [[nodiscard]] static consteval T size(const std::array<std::pair<T, T>, NS> &in) {
445 std::vector<node> vnodes{1};
446 vnodes.assign(1, node{});
447 for (auto [key, val] : in) {
448 T idx = 0;
449 for (T d = 0; d < bitslices; d++) {
450 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
451 T next = vnodes.at(idx).child[nib];
452 if (next == invalid) {
453 next = static_cast<T>(vnodes.size());
454 vnodes.at(idx).child[nib] = next;
455 vnodes.emplace_back();
456 }
457 idx = next;
458 }
459 vnodes.at(idx).child[key & child_nodes_n_mask] = val;
460 }
461 return static_cast<T>(vnodes.size());
462 }
463
464 [[nodiscard]] constexpr T lookup(T key) const {
465 T idx = 0;
466 for (T d = 0; d < bitslices; ++d) {
467 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
468 T next = nodes.at(idx).child[nib];
469 if (next == invalid) {
470 return invalid;
471 }
472 idx = next;
473 }
474 return nodes.at(idx).child[key & child_nodes_n_mask];
475 }

585 static constexpr void png_header(F &&char_out, size_t w, size_t h, size_t depth) {
586 const size_t chunkLength = 17;
587 std::array<char, chunkLength> header;
588 size_t i = 0;
589 header.at(i++) = 'I';
590 header.at(i++) = 'H';
591 header.at(i++) = 'D';
592 header.at(i++) = 'R';
593 header.at(i++) = static_cast<char>((w >> 24) & 0xFF);
594 header.at(i++) = static_cast<char>((w >> 16) & 0xFF);
595 header.at(i++) = static_cast<char>((w >> 8) & 0xFF);
596 header.at(i++) = static_cast<char>((w >> 0) & 0xFF);
597 header.at(i++) = static_cast<char>((h >> 24) & 0xFF);
598 header.at(i++) = static_cast<char>((h >> 16) & 0xFF);
599 header.at(i++) = static_cast<char>((h >> 8) & 0xFF);
600 header.at(i++) = static_cast<char>((h >> 0) & 0xFF);
601 header.at(i++) = static_cast<char>(depth);

779 template <typename PBT, size_t PBS>
780 struct palette_bitset {
781 constexpr void mark(PBT col) {
782 PBT idx = (col >> 5) & set_idx_mask;
783 set[idx] |= 1UL << (col & 0x1F);
784 }
785
786 constexpr void clear() {
787 set.fill(0);
788 }

790 [[nodiscard]] constexpr size_t genstack(std::array<PBT, PBS> &stack) const {
791 size_t count = 0;
792 for (size_t c = 0; c < PBS / 32; c++) {
793 for (size_t d = 0; d < 32; d++) {
794 if ((set[c] >> d) & 1) {
795 stack.at(count++) = static_cast<PBT>(c * 32 + d);
796 }
797 }
798 }
799 return count;
800 }
801
802 std::array<uint32_t, PBS / 32> set{};
803 static constexpr size_t set_idx_mask = (1UL << sizeof(set) / 4) - 1;
804 };

901 static constexpr void transpose8x8(std::array<uint8_t, 8> &a) {
902 // clang-format off
903 uint32_t x = (static_cast<uint32_t>(a[0]) << 24) |
904 (static_cast<uint32_t>(a[1]) << 16) |
905 (static_cast<uint32_t>(a[2]) << 8) |
906 (static_cast<uint32_t>(a[3]));
907 uint32_t y = (static_cast<uint32_t>(a[4]) << 24) |
908 (static_cast<uint32_t>(a[5]) << 16) |
909 (static_cast<uint32_t>(a[6]) << 8) |
910 (static_cast<uint32_t>(a[7]));
911
912 uint32_t t = (x ^ (x >> 7)) & 0x00AA00AA; x = x ^ t ^ (t << 7);
913 t = (y ^ (y >> 7)) & 0x00AA00AA; y = y ^ t ^ (t << 7);
914 t = (x ^ (x >> 14)) & 0x0000CCCC; x = x ^ t ^ (t << 14);
915 t = (y ^ (y >> 14)) & 0x0000CCCC; y = y ^ t ^ (t << 14);
916
917 t = ((y >> 4) & 0x0F0F0F0F) | (x & 0xF0F0F0F0);
918 y = ((x << 4) & 0xF0F0F0F0) | (y & 0x0F0F0F0F);
919
920 a[0] = static_cast<uint8_t>(t >> 24);
921 a[1] = static_cast<uint8_t>(t >> 16);
922 a[2] = static_cast<uint8_t>(t >> 8);
923 a[3] = static_cast<uint8_t>(t);
924 a[4] = static_cast<uint8_t>(y >> 24);
925 a[5] = static_cast<uint8_t>(y >> 16);
926 a[6] = static_cast<uint8_t>(y >> 8);
927 a[7] = static_cast<uint8_t>(y);
928 // clang-format on
929 }

931 static constexpr void transpose(const uint8_t *src, uint8_t *dst) {
932 std::array<uint8_t, 8> tmp;
933 size_t src_stride = ((W + 7) / 8);
934 size_t dst_stride = ((H + 7) / 8);
935 for (size_t y = 0; y < dst_stride; y++) {
936 for (size_t x = 0; x < src_stride; x++) {
937 size_t xl = std::min(size_t{8}, H - (y * 8));
938 for (size_t c = 0; c < xl; c++) {
939 tmp[c] = src[(y * 8 + c) * src_stride + x];
940 }
941 for (size_t c = xl; c < 8; c++) {
942 tmp[c] = 0;
943 }
944 transpose8x8(tmp);
945 size_t yl = std::min(size_t{8}, W - (x * 8));
946 for (size_t c = 0; c < yl; c++) {
947 dst[(x * 8 + c) * dst_stride + y] = tmp[c];
948 }
949 }
950 }

959 static constexpr void plot(std::array<uint8_t, image_size> &data, size_t x0, size_t y, uint8_t col) {
960 col &= (1UL << bits_per_pixel) - 1;

968 static constexpr void span(std::array<uint8_t, image_size> &data, size_t xl0, size_t xr0, size_t y, uint8_t col) {
969 col &= (1UL << bits_per_pixel) - 1;
970 size_t xl8 = xl0 / 8;
971 xl0 %= 8;
972 size_t xr8 = xr0 / 8;
973 xr0 %= 8;
974 size_t xs8 = xr8 - xl8;
975 uint8_t c8 =
976 static_cast<uint8_t>(col << 7 | col << 6 | col << 5 | col << 4 | col << 3 | col << 2 | col << 1 | col << 0);
977 constexpr uint8_t ml[] = {0b11111111, 0b01111111, 0b00111111, 0b00011111,
978 0b00001111, 0b00000111, 0b00000011, 0b00000001};
979 constexpr uint8_t mr[] = {0b00000000, 0b10000000, 0b11000000, 0b11100000,
980 0b11110000, 0b11111000, 0b11111100, 0b11111110};
981 uint8_t *yptr = &data.data()[y * bytes_per_line];
982 if (xs8 > 0) {
983 yptr[xl8] &= ~ml[xl0];
984 yptr[xl8] |= ml[xl0] & c8;
985 for (size_t x = xl8 + 1; x < xr8; x++) {
986 yptr[x] = c8;
987 }
988 yptr[xr8] &= ~mr[xr0];
989 yptr[xr8] |= mr[xr0] & c8;
990 } else {
991 yptr[xl8] &= ~(ml[xl0] & mr[xr0]);
992 yptr[xl8] |= (ml[xl0] & mr[xr0] & c8);
993 }
994 }
