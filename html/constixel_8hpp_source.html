53[[nodiscard]] static constexpr float fast_exp2(const float p) {
54 const float offset = (p < 0) ? 1.0f : 0.0f;
55 const float clipp = (p < -126) ? -126.0f : p;
56 const float z = clipp - static_cast<float>(static_cast<int32_t>(clipp)) + offset;
57 return std::bit_cast<float>(
58 static_cast<uint32_t>((1 << 23) * (clipp + 121.2740575f + 27.7280233f / (4.84252568f - z) - 1.49012907f * z)));
59}

61[[nodiscard]] static constexpr float fast_log2(const float x) {
62 const auto xi = std::bit_cast<uint32_t>(x);
63 const auto xf = std::bit_cast<float>((xi & 0x007FFFFF) | 0x3f000000);
64 const float y = static_cast<float>(xi) * 1.1920928955078125e-7f;
65 return y - 124.22551499f - 1.498030302f * xf - 1.72587999f / (0.3520887068f + xf);
66}

68static constexpr float fast_sqrtf(const float x) {
69 auto i = std::bit_cast<int32_t>(x);
70 const int k = i & 0x00800000;
71 float y = 0.0f;
72 if (k != 0) {
73 i = 0x5ed9d098 - (i >> 1);
74 y = std::bit_cast<float>(i);
75 y = 2.33139729f * y * ((-x * y * y) + 1.07492042f);
76 } else {
77 i = 0x5f19d352 - (i >> 1);
78 y = std::bit_cast<float>(i);
79 y = 0.82420468f * y * ((-x * y * y) + 2.14996147f);
80 }
81 const float c = x * y;
82 const float r = ((y * -c) + 1.0f);
83 y = ((0.5f * c * r) + c);
84 return y;
85}

87[[nodiscard]] static constexpr float fast_pow(const float x, const float p) {
88 return fast_exp2(p * fast_log2(x));

91[[nodiscard]] static consteval double consteval_cos(double x, int32_t terms = 10) {
92 x = x - 6.283185307179586 * static_cast<int32_t>(x / 6.283185307179586); // wrap x to [0, 2π)
93 double res = 1.0;
94 double term = 1.0;
95 const double x2 = x * x;
96 for (int32_t i = 1; i < terms; ++i) {
97 term *= -x2 / ((2 * i - 1) * (2 * i));
98 res += term;
99 }
100 return res;
101}
102
103[[nodiscard]] static consteval double consteval_sin(double x, int32_t terms = 10) {
104 x = x - 6.283185307179586 * static_cast<int32_t>(x / 6.283185307179586); // wrap x to [0, 2π)
105 double res = x;
106 double term = x;
107 const double x2 = x * x;
108 for (int32_t i = 1; i < terms; ++i) {
109 term *= -x2 / ((2 * i) * (2 * i + 1));
110 res += term;
111 }
112 return res;
113}

133[[nodiscard]] static constexpr float linear_to_srgb(float c) {
134 if (c <= 0.0031308f) {
135 return 12.92f * c;
136 }
137 if (std::is_constant_evaluated()) {
138 return 1.055f * fast_pow(c, 1.0f / 2.4f) - 0.055f;
139 }
140 return 1.055f * std::pow(c, 1.0f / 2.4f) - 0.055f;
141}

143[[nodiscard]] static constexpr float srgb_to_linear(float s) {
144 if (s <= 0.040449936f) {
145 return s / 12.92f;
146 }
147 if (std::is_constant_evaluated()) {
148 return fast_pow((s + 0.055f) / 1.055f, 2.4f);
149 }
150 return std::pow((s + 0.055f) / 1.055f, 2.4f);
151}

153[[nodiscard]] static consteval srgb oklab_to_srgb_consteval(const oklab &oklab) {
154 const double l = oklab.l;
155 const double a = oklab.a;
156 const double b = oklab.b;
157
158 const double l_ = l + 0.3963377774 * a + 0.2158037573 * b;
159 const double m_ = l - 0.1055613458 * a - 0.0638541728 * b;
160 const double s_ = l - 0.0894841775 * a - 1.2914855480 * b;
161
162 const double r = 4.0767416621 * l_ - 3.3077115913 * m_ + 0.2309699292 * s_;
163 const double g = -1.2684380046 * l_ + 2.6097574011 * m_ - 0.3413193965 * s_;
164 const double bl = -0.0041960863 * l_ - 0.7034186168 * m_ + 1.7076147031 * s_;
165
166 return {static_cast<double>(linear_to_srgb(std::max(0.0f, std::min(1.0f, static_cast<float>(r))))),
167 static_cast<double>(linear_to_srgb(std::max(0.0f, std::min(1.0f, static_cast<float>(g))))),
168 static_cast<double>(linear_to_srgb(std::max(0.0f, std::min(1.0f, static_cast<float>(bl)))))};
169}

245 [[nodiscard]] constexpr uint8_t nearest(int32_t r, int32_t g, int32_t b) const {
246 int32_t best = 0;
247 int32_t bestd = 1UL << 30;
248 for (size_t i = 0; i < pal.size(); ++i) {
249 const int32_t dr = r - static_cast<int32_t>((pal[i] >> 16) & 0xFF);
250 const int32_t dg = g - static_cast<int32_t>((pal[i] >> 8) & 0xFF);
251 const int32_t db = b - static_cast<int32_t>((pal[i] >> 0) & 0xFF);
252 const int32_t d = dr * dr + dg * dg + db * db;
253 if (d < bestd) {
254 bestd = d;
255 best = static_cast<int32_t>(i);
256 }
257 }
258 return static_cast<uint8_t>(best);
259 }

269 [[nodiscard]] constexpr uint8_t nearest_linear(float r, float g, float b) const {
270 const float epsilon = 0.0001f;
271#if defined(__ARM_NEON)
272 if (!std::is_constant_evaluated() && pal.size() >= 4) {
273 const float32x4_t vR = vdupq_n_f32(r);
274 const float32x4_t vG = vdupq_n_f32(g);
275 const float32x4_t vB = vdupq_n_f32(b);
276
277 float best = std::numeric_limits<float>::infinity();
278 std::size_t bestIdx = 0;
279
280 for (size_t i = 0; i < pal.size(); i += 4) {
281 float32x4_t dr = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3]), vR);
282 float32x4_t dg = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3 + 4]), vG);
283 float32x4_t db = vsubq_f32(vld1q_f32(&linearpal_neon[i * 3 + 8]), vB);
284
285#if defined(__aarch64__) && defined(__ARM_FEATURE_FMA)
286 float32x4_t dist = vfmaq_f32(vfmaq_f32(vmulq_f32(dr, dr), dg, dg), db, db);
287#else
288 float32x4_t dist = vaddq_f32(vaddq_f32(vmulq_f32(dr, dr), vmulq_f32(dg, dg)), vmulq_f32(db, db));
289#endif
290
291 float32x2_t lo = vget_low_f32(dist), hi = vget_high_f32(dist);
292 float d0 = vget_lane_f32(lo, 0);
293 if (d0 < best) {
294 best = d0;
295 bestIdx = i;
296 if (d0 <= epsilon) {
297 break;
298 }
299 }
300 float d1 = vget_lane_f32(lo, 1);
301 if (d1 < best) {
302 best = d1;
303 bestIdx = i + 1;
304 if (d1 <= epsilon) {
305 break;
306 }
307 }
308 float d2 = vget_lane_f32(hi, 0);
309 if (d2 < best) {
310 best = d2;
311 bestIdx = i + 2;
312 if (d2 <= epsilon) {
313 break;
314 }
315 }
316 float d3 = vget_lane_f32(hi, 1);
317 if (d3 < best) {
318 best = d3;
319 bestIdx = i + 3;
320 if (d3 <= epsilon) {
321 break;
322 }
323 }
324 }
325 return static_cast<uint8_t>(bestIdx);
326 }
327#endif // #if defined(__ARM_NEON)
328#if defined(__AVX2__)
329 if (!std::is_constant_evaluated() && pal.size() >= 8) {
330 const __m256 vR = _mm256_set1_ps(r);
331 const __m256 vG = _mm256_set1_ps(g);
332 const __m256 vB = _mm256_set1_ps(b);
333
334 float best = std::numeric_limits<float>::infinity();
335 std::uint8_t bestIdx = 0;
336
337 for (size_t i = 0; i < pal.size(); i += 8) {
338 __m256 pr = _mm256_load_ps(&linearpal_avx2[i * 3]);
339 __m256 pg = _mm256_load_ps(&linearpal_avx2[i * 3 + 8]);
340 __m256 pb = _mm256_load_ps(&linearpal_avx2[i * 3 + 16]);
341
342 __m256 dist =
343 _mm256_fmadd_ps(_mm256_sub_ps(pr, vR), _mm256_sub_ps(pr, vR),
344 _mm256_fmadd_ps(_mm256_sub_ps(pg, vG), _mm256_sub_ps(pg, vG),
345 _mm256_mul_ps(_mm256_sub_ps(pb, vB), _mm256_sub_ps(pb, vB))));
346
347 alignas(32) float d[8];
348 _mm256_store_ps(d, dist);
349
350 for (size_t lane = 0; lane < 8; lane++) {
351 if (d[lane] < best) {
352 best = d[lane];
353 bestIdx = static_cast<uint8_t>(i + lane);
354 if (d[lane] <= epsilon) {
355 return static_cast<uint8_t>(bestIdx);
356 }
357 }
358 }
359 }
360
361 return static_cast<uint8_t>(bestIdx);
362 }
363#endif // #if defined(__AVX2__)
364 size_t best = 0;
365 float bestd = 100.0f;
366 for (size_t i = 0; i < pal.size(); ++i) {
367 const float dr = r - linearpal.at(i * 3 + 0);
368 const float dg = g - linearpal.at(i * 3 + 1);
369 const float db = b - linearpal.at(i * 3 + 2);
370 const float d = dr * dr + dg * dg + db * db;
371 if (d < bestd) {
372 bestd = d;
373 best = i;
374 if (d <= epsilon) {
375 break;
376 }
377 }
378 }
379 return static_cast<uint8_t>(best);
380 }

387template <size_t N, typename T>
388class hextree {
389 static constexpr T bitslices = ((sizeof(T) * 8) / 4) - 1;
390
391 static constexpr size_t child_nodes_n = 1UL << 4;
392 static constexpr size_t child_nodes_n_mask = child_nodes_n - 1;
393
394 struct node {
395 std::array<T, child_nodes_n> child{};
396 constexpr node() {
397 for (auto &c : child) {
398 c = invalid;
399 }
400 }
401 };
402
403 public:
404 static constexpr T invalid = std::numeric_limits<T>::max();
405
406 std::array<node, N> nodes{};
407
408 [[nodiscard]] consteval size_t byte_size() const {
409 return sizeof(node) * nodes.size();
410 }
411
412 ~hextree() = default;
413 hextree(const hextree &&) = delete;
414 hextree(const hextree &) = delete;
415 hextree &operator=(const hextree &) = delete;
416 hextree &operator=(hextree &&) = delete;
417
418 explicit consteval hextree() = default;
419 template <std::size_t NS>
420 explicit consteval hextree(const std::array<std::pair<T, T>, NS> &in) {
421 nodes[0] = node{};
422 T node_cnt = 1;
423 for (auto [key, val] : in) {
424 T idx = 0;
425 for (T d = 0; d < bitslices; d++) {
426 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
427 T next = nodes.at(idx).child[nib];
428 if (next == invalid) {
429 next = node_cnt;
430 nodes.at(idx).child[nib] = next;
431 node_cnt++;
432 }
433 idx = next;
434 }
435 nodes.at(idx).child[key & child_nodes_n_mask] = val;
436 }
437 }
438
439 template <std::size_t NS>
440 [[nodiscard]] static consteval T size(const std::array<std::pair<T, T>, NS> &in) {
441 std::vector<node> vnodes{1};
442 vnodes.assign(1, node{});
443 for (auto [key, val] : in) {
444 T idx = 0;
445 for (T d = 0; d < bitslices; d++) {
446 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
447 T next = vnodes.at(idx).child[nib];
448 if (next == invalid) {
449 next = static_cast<T>(vnodes.size());
450 vnodes[idx].child[nib] = next;
451 vnodes.emplace_back();
452 }
453 idx = next;
454 }
455 vnodes[idx].child[key & child_nodes_n_mask] = val;
456 }
457 return static_cast<T>(vnodes.size());
458 }
459
460 [[nodiscard]] constexpr T lookup(T key) const {
461 T idx = 0;
462 for (T d = 0; d < bitslices; ++d) {
463 T nib = (key >> ((bitslices - d) * 4)) & child_nodes_n_mask;
464 T next = nodes.at(idx).child[nib];
465 if (next == invalid) {
466 return invalid;
467 }
468 idx = next;
469 }
470 return nodes.at(idx).child[key & child_nodes_n_mask];
471 }

597 static constexpr void png_header(F &&char_out, size_t w, size_t h, size_t depth) {
598 const size_t chunkLength = 17;
599 std::array<char, chunkLength> header{};
600 size_t i = 0;
601 header.at(i++) = 'I';
602 header.at(i++) = 'H';
603 header.at(i++) = 'D';
604 header.at(i++) = 'R';
605 header.at(i++) = static_cast<char>((w >> 24) & 0xFF);
606 header.at(i++) = static_cast<char>((w >> 16) & 0xFF);

792 template <typename PBT, size_t PBS>
793 struct palette_bitset {
794 constexpr void mark(PBT col) {
795 const PBT idx = (col >> 5) & set_idx_mask;
796 set[idx] |= 1UL << (col & 0x1F);
797 }
798
799 constexpr void clear() {
800 set.fill(0);
801 }

803 [[nodiscard]] constexpr size_t genstack(std::array<PBT, PBS> &stack) const {
804 size_t count = 0;
805 for (size_t c = 0; c < PBS / 32; c++) {
806 for (size_t d = 0; d < 32; d++) {
807 if ((set[c] >> d) & 1) {
808 stack.at(count++) = static_cast<PBT>(c * 32 + d);
809 }
810 }
811 }
812 return count;
813 }

815 std::array<uint32_t, PBS / 32> set{};
816 static constexpr size_t set_idx_mask = (1UL << sizeof(set) / 4) - 1;
817 };

913 static constexpr void transpose8x8(std::array<uint8_t, 8> &a) {
914 // clang-format off
915 uint32_t x = (static_cast<uint32_t>(a[0]) << 24) |
916 (static_cast<uint32_t>(a[1]) << 16) |
917 (static_cast<uint32_t>(a[2]) << 8) |
918 (static_cast<uint32_t>(a[3]));

920 (static_cast<uint32_t>(a[5]) << 16) |
921 (static_cast<uint32_t>(a[6]) << 8) |
922 (static_cast<uint32_t>(a[7]));
923
924 uint32_t t = (x ^ (x >> 7)) & 0x00AA00AA; x = x ^ t ^ (t << 7);
925 t = (y ^ (y >> 7)) & 0x00AA00AA; y = y ^ t ^ (t << 7);
926 t = (x ^ (x >> 14)) & 0x0000CCCC; x = x ^ t ^ (t << 14);
927 t = (y ^ (y >> 14)) & 0x0000CCCC; y = y ^ t ^ (t << 14);
928
929 t = ((y >> 4) & 0x0F0F0F0F) | (x & 0xF0F0F0F0);
930 y = ((x << 4) & 0xF0F0F0F0) | (y & 0x0F0F0F0F);
931
932 a[0] = static_cast<uint8_t>(t >> 24);
933 a[1] = static_cast<uint8_t>(t >> 16);
934 a[2] = static_cast<uint8_t>(t >> 8);
935 a[3] = static_cast<uint8_t>(t);
936 a[4] = static_cast<uint8_t>(y >> 24);
937 a[5] = static_cast<uint8_t>(y >> 16);

944 static constexpr void transpose(const uint8_t *src, uint8_t *dst) {
945 std::array<uint8_t, 8> tmp{};
946 const size_t src_stride = ((W + 7) / 8);
947 const size_t dst_stride = ((H + 7) / 8);
948 for (size_t y = 0; y < dst_stride; y++) {
949 for (size_t x = 0; x < src_stride; x++) {
950 size_t xl = std::min(size_t{8}, H - (y * 8));
951 for (size_t c = 0; c < xl; c++) {
952 if constexpr (FLIP_V) {
953 tmp[c] = src[(H - 1 - (y * 8 + c)) * src_stride + x];
954 } else {
955 tmp[c] = src[(y * 8 + c) * src_stride + x];
